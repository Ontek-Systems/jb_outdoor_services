process_form — PHP Reference

This file is a static HTML preservation of the original process_form.php server-side handler. It is kept here for reference only. On the static site, all form submissions are handled client-side via the Google Apps Script webhook in index.html and contact.html.

<?php
// 1. SECURITY: Only allow POST requests (prevents people from just typing the URL in their browser)
if ($_SERVER["REQUEST_METHOD"] !== "POST") {
    http_response_code(405);
    die("Method Not Allowed");
}

// 2. SPAM PROTECTION: Check Honeypots
// 'botcheck' is from contact.php, 'website_url' is from index.php.
// Real users can't see these, so if they are filled out, a bot did it.
if (!empty($_POST['botcheck']) || !empty($_POST['website_url'])) {
    // Silently kill the script. We pretend it worked so the bot doesn't try different tactics.
    die("Thank you for your message.");
}

// 3. RATE LIMITING: Prevent spam flooding (max 1 submission per 60 seconds per session)
session_start();
if (isset($_SESSION['last_form_submit']) && (time() - $_SESSION['last_form_submit']) < 60) {
    http_response_code(429);
    die("Error: Please wait a moment before submitting again.");
}

// 4. CSRF PROTECTION: Validate the token generated when the form page was loaded
if (
    !isset($_POST['csrf_token'], $_SESSION['csrf_token']) ||
    !hash_equals($_SESSION['csrf_token'], $_POST['csrf_token'])
) {
    http_response_code(403);
    die("Error: Security validation failed. Please refresh the page and try again.");
}
// Immediately destroy the token so it can only be used once (prevents replay attacks)
unset($_SESSION['csrf_token']);

// 5. HEADER INJECTION SANITIZER: Strip \r \n and encoded variants from any value
//    that could end up inside an email header (subject, Reply-To, From, etc.)
function sanitize_header_field(string $value): string {
    return str_replace(["\r", "\n", "\r\n", "%0a", "%0d", "%0A", "%0D", "\0"], '', $value);
}

// 6. SANITIZE HTTP_HOST: Strip non-hostname chars to prevent host-header injection
//    into the From: email header. Vercel enforces the real host, but defence-in-depth.
$safe_host = preg_replace('/[^a-zA-Z0-9.\-]/', '', $_SERVER['HTTP_HOST']);

// 7. INPUT SANITIZATION: Clean the data to prevent XSS and code injection
$name    = sanitize_header_field(isset($_POST['name'])    ? htmlspecialchars(strip_tags(trim($_POST['name'])),    ENT_QUOTES, 'UTF-8') : '');
$phone   = sanitize_header_field(isset($_POST['phone'])   ? htmlspecialchars(strip_tags(trim($_POST['phone'])),   ENT_QUOTES, 'UTF-8') : '');
$email   = sanitize_header_field(isset($_POST['email'])   ? filter_var(trim($_POST['email']), FILTER_SANITIZE_EMAIL) : '');
$message = isset($_POST['message']) ? htmlspecialchars(strip_tags(trim($_POST['message'])), ENT_QUOTES, 'UTF-8') : '';

// 8. VALIDATION: Ensure required fields aren't empty
if (empty($name) || empty($phone) || empty($message)) {
    http_response_code(400);
    die("Error: Please fill out all required fields (Name, Phone, and Message).");
}

// 9. EMAIL FORMAT VALIDATION: If they provided an email, make sure it looks valid
if (!empty($email) && !filter_var($email, FILTER_VALIDATE_EMAIL)) {
    http_response_code(400);
    die("Error: The email address provided is invalid.");
}

// 10. EMAIL PREPARATION
$to      = "mattsyardservices35@gmail.com";
$subject = sanitize_header_field("New Website Lead: Quote Request from $name");

// Build the email body
$body  = "You have received a new quote request from your website.\n\n";
$body .= "--- CUSTOMER DETAILS ---\n";
$body .= "Name: $name\n";
$body .= "Phone: $phone\n";
if (!empty($email)) {
    $body .= "Email: $email\n";
}
$body .= "\n--- PROJECT DETAILS ---\n";
$body .= "$message\n";

// Build headers — sanitized host, sanitized email, no PHP version exposed
$headers  = "From: noreply@" . $safe_host . "\r\n";
if (!empty($email)) {
    $headers .= "Reply-To: $email\r\n";
}
$headers .= "X-Mailer: WebMailer/1.0\r\n";
$headers .= "MIME-Version: 1.0\r\n";
$headers .= "Content-Type: text/plain; charset=UTF-8\r\n";

// 11. GOOGLE SHEETS: Send lead data to the connected spreadsheet via Apps Script webhook
$webhookUrl = "https://script.google.com/macros/s/AKfycbx41GRUXppT2pszgor1UeM-5-_HvfkkTBMKjC6jev0EY3ZeeBzNIttrq7_3x-RUOrdN/exec";

$sheetData = json_encode([
    "name"    => $name,
    "email"   => $email,
    "phone"   => $phone,
    "message" => $message
]);

$ch = curl_init($webhookUrl);
curl_setopt($ch, CURLOPT_POST,           true);
curl_setopt($ch, CURLOPT_POSTFIELDS,     $sheetData);
curl_setopt($ch, CURLOPT_HTTPHEADER,     ["Content-Type: application/json"]);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
curl_setopt($ch, CURLOPT_TIMEOUT,        10);
curl_exec($ch);
curl_close($ch);
// Note: we intentionally don't check the cURL response here.
// If the Sheet write fails, the visitor still gets their confirmation
// and Matt still gets the email — no submission is ever lost.

// 12. SEND EMAIL & REDIRECT
if (mail($to, $subject, $body, $headers)) {
    // Record submission time for rate limiting on next request
    $_SESSION['last_form_submit'] = time();
    // Set session flag so thank-you.php can verify the form was genuinely submitted
    $_SESSION['form_submitted'] = true;
    header("Location: /thank-you.html");
    exit;
} else {
    http_response_code(500);
    die("Error: There was a problem sending your message. Please try calling instead: +44 7480 209512");
}
?>
